cmake_minimum_required(VERSION 3.5)

project(menusearch VERSION 0.1 LANGUAGES CXX)
set(QT_VERSION_MAJOR 5)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
find_package(dbusmenu-qt5 REQUIRED)

find_package(QT NAMES Qt5 REQUIRED COMPONENTS Widgets DBus)
find_package(X11 COMPONENTS Xtst REQUIRED)
find_package(Qt${QT_VERSION_MAJOR} REQUIRED COMPONENTS Widgets DBus)
find_package(dbusmenu-qt5 REQUIRED)
set(PROJECT_SOURCES
        main.cpp
        mainwindow.cpp
        mainwindow.h
        menuimporteradaptor.h
        menuimporteradaptor.cpp
        menuimporter.h menuimporter.cpp
        menux11.h menux11.cpp
)

if(${QT_VERSION_MAJOR} GREATER_EQUAL 6)
    qt_add_executable(menusearch
        MANUAL_FINALIZATION
        ${PROJECT_SOURCES}
    )
# Define target properties for Android with Qt 6 as:
#    set_property(TARGET menusearch APPEND PROPERTY QT_ANDROID_PACKAGE_SOURCE_DIR
#                 ${CMAKE_CURRENT_SOURCE_DIR}/android)
# For more information, see https://doc.qt.io/qt-6/qt-add-executable.html#target-creation
else()
    if(ANDROID)
        add_library(menusearch SHARED
            ${PROJECT_SOURCES}
        )
# Define properties for Android with Qt 5 after find_package() calls as:
#    set(ANDROID_PACKAGE_SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/android")
    else()
        add_executable(menusearch
            ${PROJECT_SOURCES}


        )
    endif()
endif()
target_link_libraries(menusearch PRIVATE Qt${QT_VERSION_MAJOR}::Widgets Qt${QT_VERSION_MAJOR}::DBus dbusmenu-qt5 ${X11_Xtst_LIB} ${X11_X11_LIB} -lpthread)

# Qt for iOS sets MACOSX_BUNDLE_GUI_IDENTIFIER automatically since Qt 6.1.
# If you are developing for iOS or macOS you should consider setting an
# explicit, fixed bundle identifier manually though.
if(${QT_VERSION} VERSION_LESS 6.1.0)
  set(BUNDLE_ID_OPTION MACOSX_BUNDLE_GUI_IDENTIFIER com.example.menusearch)
endif()
set_target_properties(menusearch PROPERTIES
    ${BUNDLE_ID_OPTION}
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

include(GNUInstallDirs)
target_include_directories(menusearch PUBLIC ${dbusmenu-qt6_INCLUDE_DIRS} ${X11_Xtst_INCLUDE_DIRS} ${X11_X11_INCLUDE_DIRS})

install(TARGETS menusearch
    BUNDLE DESTINATION .
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)

target_include_directories(menusearch PUBLIC ${dbusmenu-qt6_INCLUDE_DIRS} ${X11_Xtst_INCLUDE_DIRS} ${X11_X11_INCLUDE_DIRS})
if(QT_VERSION_MAJOR EQUAL 6)
    qt_finalize_executable(menusearch)
endif()
